# Boucle sur chaque classe dans y_train
for label in np.unique(y_train):
    # Crer un classificateur pour cette classe spcifique
    clf_label = RandomForestClassifier(n_estimators=100, random_state=42)
    
    # Slectionner les exemples appartenant  cette classe
    X_train_label = X_train[y_train == label]
    
    # Entraner le classificateur sur les exemples de cette classe uniquement
    clf_label.fit(X_train_label, np.ones(X_train_label.shape[0]))
    
    # Slectionner les meilleures caractristiques pour cette classe spcifique
    sfm = SelectFromModel(clf_label, threshold='median')
    sfm.fit(X_train_label, np.ones(X_train_label.shape[0]))
    
    # Rcuprer les indices des caractristiques slectionnes
    selected_indices = sfm.get_support(indices=True)
